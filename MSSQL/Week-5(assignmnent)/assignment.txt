-- query-9 ( Find the duration (in days) of each project using DATEDIFF.)

select * from Projects
select name as project_name,DATEDIFF(day,startdate,enddate) as number_of_day from Projects


-- query-10 ( Add 10 days to each project’s EndDate using DATEADD.)

select * from Projects
select name as project_name,DATEADD(day,10,enddate) as new_enddate from Projects


-- query-11 ( Find how many days are left until each project ends. )

select * from Projects
select name as project_name,DATEDIFF(day,getdate(),EndDate) as new_enddate from Projects

--3. For employees (from Employees table above), categorize them:

--* 	If Dept = HR → show Human Resources
--* 	Else → Other)

    Name,
    Department,
    CASE
        WHEN Department = 'IT' THEN 'Tech Team'

    END AS Dept_categoty
